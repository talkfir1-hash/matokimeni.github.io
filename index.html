  var cart = [];


  function getSelected(name) {

    var els = document.querySelectorAll('input[name="' + name + '"]:checked');

    if (els.length === 0) return null;

    return Array.from(els).map(function(el) { return el.value; }).join(', ');

  }


  function getCakeDetails() {

    var types = getSelected('cake');

    var event = document.getElementById('cakeEvent').value;

    var color = document.getElementById('cakeColor').value;

    if (!types) return null;

    return types + (event ? ' | אירוע: ' + event : '') + (design ? ' | עיצוב: ' + design : '') + (color ? ' | צבע: ' + color : '');

  }

    var details = document.getElementById('packageDetails').value;

    if (!content) return null;

    return content + (details ? ' | ' + details : '');

  }


  function addToCart(type, details) {

    if (!details) { alert('אנא בחרי סוג לפני הוספה לעגלה'); return; }

    cart.push({ type: type, details: details });

    updateCartUI();

    openCart();

  }


  function updateCartUI() {

    document.getElementById('cartCount').textContent = cart.length;

    var el = document.getElementById('cartItems');

    var footer = document.getElementById('cartFooter');

    if (cart.length === 0) {

      el.innerHTML = '<p class="cart-empty">העגלה ריקה עדיין 🛒</p>';

      footer.style.display = 'none';

      return;

    }

    footer.style.display = 'block';

    el.innerHTML = cart.map(function(item, i) {

      return '<div class="cart-item"><div class="cart-item-name"><strong>' + item.type + '</strong><br><span style="font-size:0.8rem;color:#999;">' + item.details + '</span></div><button class="cart-item-remove" onclick="removeFromCart(' + i + ')">✕</button></div>';

    }).join('');

  }


  function removeFromCart(i) {

    cart.splice(i, 1);

    updateCartUI();

  }


  function openCart() {

    document.getElementById('cartModal').classList.add('open');

    document.getElementById('cartOverlay').classList.add('open');

  }


  function closeCart() {

    document.getElementById('cartModal').classList.remove('open');

    document.getElementById('cartOverlay').classList.remove('open');

  }


  function openDetailsModal() {

    closeCart();

    document.getElementById('detailsOverlay').classList.add('open');

  }


  function closeDetailsModal() {

    document.getElementById('detailsOverlay').classList.remove('open');

  }


  function toggleDelivery() {

    var val = document.querySelector('input[name="delivery"]:checked');

    document.getElementById('addressField').style.display = val && val.value === 'משלוח' ? 'block' : 'none';

  }


  function sendFinalOrder() {

    var name = document.getElementById('detailsName').value;

    var phone = document.getElementById('detailsPhone').value;

    var delivery = document.querySelector('input[name="delivery"]:checked');

    var address = document.getElementById('detailsAddress').value;

    var date = document.getElementById('detailsDate').value;

    var allergies = document.getElementById('detailsAllergies').value;

    var notes = document.getElementById('detailsNotes').value;


    if (!name || !phone) { alert('אנא מלאי שם ומספר טלפון'); return; }

    if (!delivery) { alert('אנא בחרי איסוף עצמי או משלוח'); return; }

    if (delivery.value === 'משלוח' && !address) { alert('אנא מלאי כתובת למשלוח'); return; }


    var msg = 'הזמנה חדשה ממתוק ממני!\n\n';

    msg += 'שם: ' + name + '\n';

    msg += 'טלפון: ' + phone + '\n';

    msg += 'אופן קבלה: ' + delivery.value + (address ? ' — ' + address : '') + '\n';

    if (date) msg += 'מועד רצוי: ' + new Date(date).toLocaleString('he-IL') + '\n';

    if (allergies) msg += 'אלרגנים: ' + allergies + '\n';

    if (notes) msg += 'הערות: ' + notes + '\n';

    msg += '\nפרטי הזמנה:\n';

    cart.forEach(function(item, i) {

      msg += (i+1) + '. ' + item.type + ': ' + item.details + '\n';

    });


    var url = 'https://wa.me/972549915744?text=' + encodeURIComponent(msg);

    window.open(url, '_blank');

    closeDetailsModal();

    cart = [];

    updateCartUI();

  }
